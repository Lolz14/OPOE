172  void update_characteristic_fn() const {
173  ComplexStoringVector res(N_);
174  this->sde_model_->characteristic_fn(this->ttm_, v_ - SDE::ImaginaryUnit<R>, res);
175 
176  auto exp_term = (SDE::ImaginaryUnit<R> * this->rate_ * v_ * this->ttm_).exp();
177  ComplexArray numerator = res.array() - std::complex<R>(1.0, 0.0);
178  ComplexArray denominator = SDE::ImaginaryUnit<R> * v_ * (v_ * SDE::ImaginaryUnit<R> + std::complex<R>(1.0, 0.0));
179  ComplexArray Z_k = exp_term.array() * numerator.array() / denominator.array();

183  ComplexArray lin = Array::LinSpaced(N_, 0, N_ - 1).template cast<std::complex<R>>();
184  ComplexArray x = w_.template cast<std::complex<R>>() * std::complex<R>(eta_, 0) *
185  (SDE::ImaginaryUnit<R> * M_PI * lin).array().exp() * Z_k.array();

227  requires GetterFunction<Getter, R> && SetterFunction<Setter, R>
