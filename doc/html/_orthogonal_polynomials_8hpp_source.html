154  void generatePolynomials() {
155  #pragma omp parallel for simd
156  for (unsigned int i = 0; i < N + 2; ++i) {
157  polynomials[i] = PolyType(StoringArray::Zero(N + 1));
158  }
159 
160  // P₀(x) is implicitly zero.
161  // P₁(x) = 1 (coefficient of x⁰ is 1)
162  if (N + 1 > 0) {
163  polynomials[1].get_coeff()(0) = 1.0/std::sqrt(recurrence_coeffs.beta_0(0.0));
164  //polynomials[1].get_coeff()(0) = 1.0;
165  }
166  H.col(0) = polynomials[1].get_coeff();
167 
168  if constexpr (N == 0) {
169  // Pre-calculate and store alpha_0 and beta_0 if needed elsewhere
170  if (N + 1 > 0) { // Need alpha_0 for J matrix
171  alphas[0] = recurrence_coeffs.alpha_k(0);

179  // --- Sequential Recurrence Calculation ---
180 
181  // Temporary storage for the coefficients of P_{k+1}
182  // Allocated once outside the loop to avoid repeated allocations.
183  StoringArray P_kp1_coeffs(N + 1);
184 
185  // Calculate P₂, P₃, ..., P_{N+1} sequentially
186  for (unsigned int k = 1; k <= N; ++k) {
187  // Calculate recurrence coefficients for this step
188  const R alpha_km1 = recurrence_coeffs.alpha_k(k - 1); // α_{k-1}
189  const R beta_km1 = (k == 1) ? recurrence_coeffs.beta_0(0.0) // β₀
190  : recurrence_coeffs.beta_k(k - 1); // β_{k-1}
191  const R beta_kp1 = recurrence_coeffs.beta_k(k); // β_{k}
192 
193  // Store coefficients for later use (e.g., Jacobi matrix, evaluation)
194  alphas[k - 1] = alpha_km1;
195  betas[k - 1] = beta_km1; // Store β₀, β₁, ..., β_{N-1}
196 
197  // Get references to coefficients of P_k and P_{k-1}
198  const auto& P_k_coeffs = polynomials[k].get_coeff();
199  const auto& P_km1_coeffs = polynomials[k - 1].get_coeff();
200 
201  // --- Calculation using Eigen's vectorized operations ---
202  // P_kp1 = (x * P_k) - alpha_km1 * P_k - beta_km1 * P_km1
203 
204  // 1. Compute x * P_k (results in higher degree, store temporarily)
205  P_kp1_coeffs.setZero(); // Clear previous iteration's result
206  // Shift P_k's coefficients: P_k[i] -> (x*P_k)[i+1]
207  // Copies P_k[0..N-1] to P_kp1_coeffs[1..N]
208  P_kp1_coeffs.segment(1, N) = P_k_coeffs.head(N);
209 
210  // 2. Subtract alpha_km1 * P_k
211  P_kp1_coeffs -= alpha_km1 * P_k_coeffs;
212 
213  // 3. Subtract beta_km1 * P_{k-1}
214  // P_kp1_coeffs -= beta_km1 * P_km1_coeffs;
215 
216  P_kp1_coeffs -= std::sqrt(beta_km1) * P_km1_coeffs;
217  P_kp1_coeffs /= std::sqrt(beta_kp1); // Normalize by sqrt(beta_k)
218  // --- End of Eigen calculation ---
219 
220  // Store the final coefficients for P_{k+1}
221  H.col(k) = P_kp1_coeffs;
222  polynomials[k + 1].set_coeff(P_kp1_coeffs);
223  }
224 
225 
226  // Store coefficients for later use (e.g., Jacobi matrix, evaluation)
227  alphas[N] = recurrence_coeffs.alpha_k(N);
228  betas[N] = recurrence_coeffs.beta_k(N);
229  }

294  void buildJacobiMatrix() {
295  // Fill the Jacobi matrix with recurrence coefficients
296  Triplets triplets(N + 1);
297  const int expectedNonZeros = 3 * N + 1; // Main diagonal + upper + lower
298  triplets.reserve(expectedNonZeros);
299 
300 
301  #pragma omp parallel
302  {
303  // Each thread maintains its own local vector of triplets
304  Triplets localTriplets;
305  localTriplets.reserve(expectedNonZeros / omp_get_num_threads());
306 
307  #pragma omp for nowait
308  for(unsigned int i = 0; i < N + 1; ++i) {
309  // Main diagonal
310  localTriplets.emplace_back(i, i, alphas[i]);
311 
312  // Lower and upper diagonals
313  if(i > 0) {
314  localTriplets.emplace_back(i, i-1, std::sqrt(betas[i])); // Lower
315  localTriplets.emplace_back(i-1, i, std::sqrt(betas[i])); // Upper
316  }
317  }
318 
319  // Merge local triplets into global vector
320  #pragma omp critical
321  {
322  triplets.insert(triplets.end(),
323  localTriplets.begin(),
324  localTriplets.end());
325  }
326  }
327 
328  // Set matrix from triplets
329  J.resize(N + 1, N + 1);
330  J.setFromTriplets(triplets.begin(), triplets.end());
331  J.makeCompressed();
332  };
