166  constexpr size_t number_of_dimensions = sizeof...(dims_args);

173

175  return traits::DataType::StoringVector::NullaryExpr(vector_size, [&]() -> T {

182  // Extract matrix rows and columns from the tuple

189  });

211 std::tuple<int, T, traits::DataType::StoringVector> argmin(
212  const traits::DataType::StoringMatrix& matrix,
213  const traits::DataType::StoringVector& target_vector) {
214 
215  // --- Input Validations ---
216  if (matrix.cols() == 0) {
217  std::cerr << "Error: Matrix has no columns." << std::endl;
218  return {-1, std::numeric_limits<T>::infinity(), traits::DataType::StoringVector()};
219  }
220  if (matrix.rows() != target_vector.size()) {
221  std::cerr << "Error: Number of rows in matrix (" << matrix.rows()
222  << ") does not match size of target vector (" << target_vector.size()
223  << ")." << std::endl;
224  return {-1, std::numeric_limits<T>::infinity(), traits::DataType::StoringVector()};
225  }
226 
227  // Compute the L2 distances between each column and the target vector
228  traits::DataType::StoringVector l2_distances = (matrix.colwise() - target_vector).colwise().norm();
229 
230  // Find the minimum L2 distance and its index
231  Eigen::Index min_col_index;
232  T min_distance_value = l2_distances.minCoeff(&min_col_index);
233 
234  return {static_cast<int>(min_col_index), min_distance_value, l2_distances};
235 }

251 template<typename T = traits::DataType::PolynomialField, int Dim>
252 T calculate_overall_distortion(
253  const traits::DataType::StoringMatrix& current_centroids,
254  int num_test_samples, // Number of random samples to generate
255  std::mt19937& generator,
256  std::normal_distribution<T>& distribution) {
257 
258  // Validate input
259  if (current_centroids.cols() == 0 || num_test_samples == 0) {
260  if (current_centroids.cols() == 0) std::cerr << "Distortion calc error: No centroids." << std::endl;
261  return std::numeric_limits<T>::infinity();
262  }
263  if (current_centroids.rows() != Dim) {
264  std::cerr << "Distortion calc error: Centroid dimension mismatch." << std::endl;
265  return std::numeric_limits<T>::infinity();
266  }
267 
268  T total_squared_error = static_cast<T>(0.0);
269  int valid_samples_for_distortion = 0;
270 
271  // Generate random samples and accumulate squared error to closest centroid
272  for (int i = 0; i < num_test_samples; ++i) {
273  // Generate a single, fresh test sample
274  traits::DataType::StoringVector test_sample =
275  sampler<T>(generator, distribution, Dim);
276 
277  // Find the closest centroid and its L2 distance
278  auto closest_info_for_test_sample = argmin<T>(current_centroids, test_sample);
279 
280  if (std::get<0>(closest_info_for_test_sample) != -1) { // Valid centroid found
281  T min_l2_distance = std::get<1>(closest_info_for_test_sample);
282  total_squared_error += (min_l2_distance * min_l2_distance); // Accumulate squared L2 distance
283  valid_samples_for_distortion++;
284  }
285  }
286 
287  // Compute RMSE if there are valid samples
288  if (valid_samples_for_distortion > 0) {
289  return std::sqrt(total_squared_error / static_cast<T>(valid_samples_for_distortion));
290  }
291 
292  return std::numeric_limits<T>::infinity(); // Return infinity if no valid samples
293 }

399  const StoringVector& axx,

409  const StoringMatrix D2mono = build_D2mono<T>(Nx);
410

413  const StoringMatrix D_H2 = dec.solve(D2mono * H);

417  const StoringMatrix Bxx = poly_to_M<T>(axx, Nv);

420  const StoringMatrix Bvv = poly_to_M<T>(avv, Nv);

423  const StoringMatrix D2vmono = build_D2mono<T>(Nv);

429  G_full.noalias() += 0.5 * Eigen::kroneckerProduct(Bxx, D_H2).eval();

431  G_full.noalias() += Eigen::kroneckerProduct((Bxv * Dvmono).eval(), D_H).eval();

491 StoringVector build_h_vals(const StoringMatrix& H, // (N+1)x(N+1) Hermite basis matrix
492  const std::vector<std::pair<int,int>>& E, // vector of (m,n) pairs
493  unsigned int N, // polynomial base degree
494  Scalar X0, Scalar V0){
495  const int M = static_cast<int>(E.size());
496 
497  // --- Step 1: monomials of X0 up to N
498  StoringVector monoms(N+1);
499  monoms(0) = Scalar(1);
500  for (unsigned int k = 1; k <= N; ++k) {
501  monoms(k) = monoms(k-1) * X0;
502  }
503 
504  // --- Step 2: Hermite evaluation using H matrix
505  // H is (N+1)x(N+1), monoms is (N+1)x1
506  // Hvec = H * monoms, size (N+1)x1
507  StoringVector Hvec = H.transpose() * monoms;
508 
509  // --- Step 3: powers of V0 up to max m in E
510  unsigned int max_m = 0;
511  for (auto [m, n] : E) if ((unsigned int)m > max_m) max_m = m;
512  StoringVector Vpowers(max_m+1);
513  Vpowers(0) = Scalar(1);
514  for (unsigned int k = 1; k <= max_m; ++k) {
515  Vpowers(k) = Vpowers(k-1) * V0;
516  }
517 
518  // --- Step 4: assemble h_vals
519  StoringVector h_vals(M);
520  for (int i = 0; i < M; ++i) {
521  const auto [m, n] = E[i];
522  h_vals(i) = Vpowers(m) * Hvec(n);
523  }
524 
525  return h_vals;
