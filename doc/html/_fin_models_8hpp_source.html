634  virtual SDEMatrix generator_G(std::vector<std::pair<int,int>> E, int N, T sigma) const {
635 
636  const int M = static_cast<int>(E.size());
637 
638  // Fast index lookup
639  Utils::IndexMap idx;
640  idx.reserve(M);
641  for (int i = 0; i < M; ++i) {
642  const auto& [m, n] = E[i];
643  idx.emplace(key(m, n), i);
644  }
645 
646  SDEMatrix G = SDEMatrix::Zero(M, M);
647 
648  const auto q = params_.sv_vol_exponent;
649  const auto p = params_.asset_vol_exponent;
650  const auto kappa = params_.sv_kappa;
651  const auto theta = params_.sv_theta;
652  const auto rho = params_.correlation;
653  const auto r = params_.asset_drift_const;
654  const auto sigma_v = params_.sv_sigma;
655 
656  auto add_entry = [&](int row_m, int row_n, int col, T value) {
657  if (row_m < 0 || row_n < 0) return;
658  if (row_m + row_n > N) return; // out of basis range
659  auto it = idx.find(key(row_m, row_n));
660  if (it != idx.end()) {
661  G(it->second, col) += value; // Since p and q are dynamic, the same cell may be called multiple times, so we must add to the existing value
662  }
663  };
664 
665  for (int col = 0; col < M; ++col) {
666  const auto [m, n] = E[col];
667 
668  // 1) h_{m-2+2q, n}
669  if (m >= 2 - 2 * q) {
670 
671  add_entry(m-2 + 2 * q, n, col, sigma_v * sigma_v * 0.5 * m * (m - 1));
672  }
673  // 2) h_{m-1+p+q, n-1}
674  if (m >= 1 - p - q && n >= 1) {
675 
676  add_entry(m - 1 + p + q, n-1, col, rho * sigma_v * m * std::sqrt((T)n) / sigma);
677  }
678  // 3) h_{m-1, n}
679  if (m >= 1) {
680 
681  add_entry(m - 1, n, col, m * kappa * theta);
682  }
683  // 4) h_{m, n-1}
684  if (n >= 1) {
685 
686  add_entry(m, n-1, col, std::sqrt((T)n) / sigma * r);
687  }
688  // 5) h_{m+2p, n-2}
689  if (n >= 2) {
690 
691  add_entry(m + 2 * p, n-2, col, std ::sqrt((T)n * (n - 1)) / (2.0 * sigma * sigma));
692  }
693  // 6) h_{m, n}
694  add_entry(m, n, col, -m * kappa);
695 
696  // 7) h_{m+2p, n-1}
697  if (n >= 1) {
698 
699  add_entry(m + 2 * p, n-1, col, -0.5 * std::sqrt((T)n) / sigma);
700  }
701  }

703  return G;
