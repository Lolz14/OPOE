182  static DensityInterval<R> get_safe_support(const BoostDist& dist) {
183  try {
184  // Use boost::math::support to get the theoretical support interval
185  auto support_pair = boost::math::support(dist);
186  R lower = support_pair.first;
187  R upper = support_pair.second;
188  return DensityInterval<R>(lower, upper);
189 
190  } catch (const std::exception& e) {
191  // If boost::math::support fails (e.g., Cauchy distribution), default to (-inf, inf)
192  std::cerr << "Warning: Could not determine finite support for distribution "
193  << typeid(dist).name() << ". Using (-inf, inf). Reason: " << e.what() << std::endl;
194  return DensityInterval<R>(); // Default constructor gives (-inf, inf)
195  } catch (...) {
196  std::cerr << "Warning: Unknown error determining support for distribution "
197  << typeid(dist).name() << ". Using (-inf, inf)." << std::endl;
198  return DensityInterval<R>();
199  }
